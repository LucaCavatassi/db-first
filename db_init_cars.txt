CARS
_____________________________

id                          - UNSIGNED INT - PRIMARY_KEY (NOT NULL, UNIQUE) - AUTO_INCREMENT
marchio                     - VARCHAR(255) - NOT NULL
modello                     - VARCHAR(255) - NOT NULL
anno                        - YEAR - NOT NULL
chilometraggio              - UNSIGNED INT - NOT NULL
tipo di carburante          - VARCHAR(25) - NOT NULL
tipo di cambio              - VARCHAR(25)
specifiche del motore       - VARCHAR(255)
prezzo                      - DECIMAL(10, 2) - NOT NULL
posti a sedere              - UNSIGNED TINY INT
porte                       - UNSIGNED TINY INT 
consumi                     - VARCHAR(25)
emissioni                   - VARCHAR(25)
optional                    - VARCHAR(255)
descrizione                 - TEXT
